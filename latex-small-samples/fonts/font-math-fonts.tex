\documentclass[a5paper]{article}

\usepackage{../chc-exa-style}

\usepackage{amsfonts}



\begin{document}


\note{Blackboard bolds (bb) benötigen \texttt{\textbackslash usepackage\{amsfonts\}}}

\note{\texttt{mathbb} in Command Form}

\begin{example}
$\mathbb{R} \neq \mathbb{C}$
\end{example}

\begin{example}
$\mathbb{R}^+_0 \neq \mathbb{N}_0$
\end{example}

\note{\texttt{Bbb} in Declaration Form}

\begin{example}
${\Bbb R}^+_0 \neq \Bbb{N}_0$
\end{example}



\note{Blackboard bolds (bb) sind nur für Großbuchstaben definiert.
Kleinbuchstaben verweisen auf Symbole, anders angesteuert werden sollten.}

\begin{example}
$\mathbb{r} = \mathbb{f}$
\end{example}

\bigskip

\note{Calligraphische Großbuchstaben}

\begin{example}
$\mathcal{A}=\mathcal{B}$
\end{example}

\bigskip
