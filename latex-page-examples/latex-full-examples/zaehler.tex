% Abschaltbefehl muß allein auf einer Zeile stehen

\newcounter{foo}        % Definieren neuen Zähler
\setcounter{foo}{5}     % Initialisiere neuen Zähler

\begin{itemize}
\item \verb|\arabic{foo}| ergibt arabische Zahl \arabic{foo}
\item \verb|\roman{foo}| ergibt kleine römische Zahl \roman{foo} 
\item \verb|\Roman{foo}| ergibt große römische Zahl \Roman{foo}
\item \verb|\Alph{foo}| ergibt große alphabetische Position \Alph{foo}
\item \verb|\alph{foo}| ergibt kleine alphabetische Position \alph{foo}
\end{itemize}
